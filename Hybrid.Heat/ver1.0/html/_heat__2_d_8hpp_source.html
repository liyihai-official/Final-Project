188template <typename T>
189 inline T
190 Heat_2D<T>::update_ping_pong()
191 {
192 T diff {0.0};
193 size_type i {1}, j {1};
194 for (i=1; i < in.topology().__local_shape[0] - 1; ++i)
195 {
196 for (j=1; j < in.topology().__local_shape[1] - 1; ++j)
197 {
198 T current {in(i,j)};
199 out(i,j) =
200 this->weights[0] * (in(i-1,j) + in(i+1,j))
201 + this->weights[1] * (in(i,j-1) + in(i,j+1))
202 + current * (
203 this->diags[0]*this->weights[0]
204 + this->diags[1]*this->weights[1]
205 );
206
207 diff += std::pow(current - out(i,j), 2);
208 }
209 }
210
211 return diff;
212 }

412 in.topology().comm_cart, &reqs[req_cnt++]);
413
414
415 flag = 1; dim = 1;
416 n_size = in.topology().__local_shape[dim];
417 // Send and receive for the second dimension
418 MPI_Irecv(&in(1, n_size-1), 1, in.topology().halos[dim], in.topology().nbr_dest[dim], flag,
419 in.topology().comm_cart, &reqs[req_cnt++]);

428
429

440template <typename T>
441 Integer
442 Heat_2D<T>::solve_pure_mpi(const T tol, const Integer nsteps, const Integer root)
443 {
444 T ldiff {0.0}, gdiff {0.0}, time {0}; MPI_Datatype DiffType {mpi::get_mpi_type<T>()};
445 Integer iter {1};
446
447#ifndef NDEBUG
448FINAL_PROJECT_MPI_ASSERT_GLOBAL((BC_2D != nullptr && IC_2D != nullptr));
449FINAL_PROJECT_ASSERT(BC_2D->isSetUpBC && IC_2D->isSetUpInit);
450#endif
451
452#ifndef NDEBUG
453{
454 if (root == in.topology().rank)
455 {
456 std::cout << 2 << " Dimension Simulation Parameters: " << std::endl;
457 std::cout << "\tRows: " << in.topology().__local_shape[0]-2
458 << "\n\tColumns: " << in.topology().__local_shape[1]-2 << std::endl;
459 std::cout << "\tTime steps: " << nsteps << std::endl;
460 std::cout << "\tTolerance: " << tol << std::endl;
461
462 std::cout << "MPI Parameters: " << std::endl;
463 std::cout << "\tNumber of MPI Processes: " << in.topology().num_procs << std::endl;
464 std::cout << "\tRoot Process: " << root << std::endl;
465
466 std::cout << "Heat Parameters: " << std::endl;
467 std::cout << "\tCoefficient: " << this->coff << "\n"
468 << "\tTime resolution: " << this->dt << "\n"
469 << "\tWeights: " << this->weights[0] << ", "
470 << this->weights[1] << "\n"
471 << "\tdxs: " << this->dxs[0] << ", "
472 << this->dxs[1] << std::endl;
473 }
474}
475#endif // end NDEBUG
476
477
478 Double t0 {MPI_Wtime()};
479 for (iter = 1; iter < nsteps; ++iter)
480 {
481 time = iter*this->dt;
482 exchange_ping_pong_I();
483 ldiff = update_ping_pong();
484 BC_2D->UpdateBC(*this, time);
485 MPI_Allreduce(&ldiff, &gdiff, 1, DiffType, MPI_SUM, in.topology().comm_cart);
486
487#ifndef NDEBUG
488mpi::Gather(gather, in, root);
489if (in.topology().rank == root)
490{
491 std::cout << std::fixed << std::setprecision(13) << std::setw(15) << gdiff
492 << std::endl;
493}
494#endif
495
496 switch_in_out();
497
498 if (gdiff <= tol) {
499 converge = true;
500 break;
501 }
502 }
503 Double t1 {MPI_Wtime() - t0};
504
505 if (converge)
506 {
507 Double total {0};
508 mpi::Gather(gather, in, root);
509 MPI_Reduce(&t1, &total, 1, MPI_DOUBLE, MPI_MAX, root, in.topology().comm_cart);
510 if (root == in.topology().rank)
511 std::cout << "Total Converge time: " << total << "\n"
512 << "Iterations: " << iter << std::endl;
513 } else {
514 if (in.topology().rank == root) std::cout << "Fail to converge" << std::endl;
515 }
516
517#ifndef NDEBUG // Gather
518mpi::Gather(gather, in, root);
519#endif
520
521 return iter;
522 }
