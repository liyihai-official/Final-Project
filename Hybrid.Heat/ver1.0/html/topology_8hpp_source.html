138template <typename T, size_type NumD>
139 inline
140 Cartesian<T, NumD>::Cartesian( __array_shape & global_shape, environment & env )
141: __global_shape(global_shape), __local_shape(global_shape)
142 {
143 FINAL_PROJECT_ASSERT_MSG((NumD <= 4), "Number of dimension if out of range.");
144
146auto Decomp = [](
147 const Integer n, const Integer prob_size, const Integer rank,
148 Integer & s, Integer & e)
149{
150 Integer n_loc {n / prob_size}, deficit {n % prob_size};
151
152 s = rank * n_loc + 1;
153 s += ((rank < deficit) ? rank : deficit);
154
155 if (rank < deficit) ++n_loc;
156 e = s + n_loc - 1;
157
158 if (e > n || rank == prob_size - 1) e = n;
159 return 0;
160};
161
162 Integer i {0};
163 dimension = static_cast<Integer>(NumD); // 0 < Num < 4
164 std::array<Integer, NumD> array_size, array_sub_size, array_starts = {0};
165
166 MPI_Dims_create(env.size(), dimension, dims.data());
167 MPI_Cart_create(env.comm(), dimension, dims.data(), periods.data(), 1, &comm_cart);
168 MPI_Comm_size(comm_cart, &num_procs);
169
170 MPI_Comm_rank(comm_cart, &rank);
171 MPI_Cart_coords(comm_cart, rank, dimension, coordinates.data());
172
173 for (i = 0; i < dimension; ++i)
174 {
175 MPI_Cart_shift(comm_cart, i, 1, &nbr_src[i], &nbr_dest[i]);
176
177
178 Decomp(__global_shape[i]-2, dims[i], coordinates[i], starts[i], ends[i]);
179 __local_shape[i] = ends[i] - starts[i] + 1 + 2; // Include halos
180 array_size[i] = __local_shape[i];
181 array_sub_size[i] = array_size[i] - 2;
182FINAL_PROJECT_MPI_ASSERT_GLOBAL((array_sub_size[i] > 0)); // sub size must bigger than 0
183 }
184
185 for (i = 0; i < dimension; ++i)
186 {
187 auto temp = array_sub_size[i];
188 array_sub_size[i] = 1;
189MPI_Type_create_subarray( dimension, array_size.data(),
190 array_sub_size.data(),
191 array_starts.data(),
192 MPI_ORDER_C, value_type, &halos[i]);
193
194MPI_Type_commit(&halos[i]);
195 array_sub_size[i] = temp;
196 }
197 }
