233template <class T, size_type NumD>
234 inline
235 void
236 array_base<T, NumD>::saveToBinary(const String & filename)
237 const
238{
239 std::ofstream ofs(filename, std::ios::binary);
240 FINAL_PROJECT_ASSERT_MSG(ofs, "Cannot Open File");
241
242 for (size_type i = 0 ; i < NumD; ++i)
243 {
244 Dworld temp {body->__shape[i]};
245 ofs.write(reinterpret_cast<const Char*>(&temp), sizeof((temp)));
246 }
247
248 ofs.write(reinterpret_cast<const Char*>(
249 body->begin()), body->size() * sizeof(T)
250 );
251}

257template <class T, size_type NumD>
258 inline
259 void
260 array_base<T, NumD>::loadFromBinary(const String & filename)
261 {
262 std::ifstream ifs(filename, std::ios::binary);
263 FINAL_PROJECT_ASSERT_MSG(ifs, "Cannot Open File");
264
265 std::vector<size_type> shape_loader(NumD);
266
267 for (size_type i = 0; i < NumD; ++i)
268 {
269 Dworld temp {0};
270 ifs.read(reinterpret_cast<Char*>(&temp), sizeof(temp));
271 shape_loader[i] = temp;
272 }
273
274 std::unique_ptr<array> loader { std::make_unique<array>(array_shape(shape_loader))};
275
276 size_type total_size {1};
277 for (size_type i = 0; i < NumD; ++i)
278 total_size *= loader->__shape[i];
279
280 ifs.read(reinterpret_cast<Char*>(loader->begin()), total_size * sizeof(T));
281 body.swap(loader);
282
283 }

464template <typename T, size_type NumD>
465void Gather(
466 multi_array::array_base<T, NumD> &gather,
467 array_Cart<T, NumD> &loc,
468 const Integer root)
469{
470 MPI_Datatype buf_block, value_type{get_mpi_type<T>()};
471 const Integer num_procs{loc.topology().num_procs}, dimension{loc.topology().dimension};
472
473 Integer pid, dim, back{0};
474 Integer s_list[NumD][num_procs], n_list[NumD][num_procs];
475 std::array<Integer, NumD> Ns, starts_cpy, array_sizes, array_subsizes, array_starts, indexes;
476
477 // Gather basic information of sending entities
478 if (num_procs == 1)
479 {
480 ++back;
481 } // For handling the special case that there is only 1 process.
482 array_starts.fill(0), indexes.fill(1);
483
484 for (dim = 0; dim < dimension; ++dim)
485 {
486 Ns[dim] = loc.topology().__local_shape[dim] - 2;
487 starts_cpy[dim] = loc.topology().starts[dim];
488
489 if (starts_cpy[dim] == 1)
490 {
491 --starts_cpy[dim];
492 --indexes[dim];
493 ++Ns[dim];
494 }
495
496 if (loc.topology().ends[dim] == loc.topology().__global_shape[dim] - 2)
497 ++Ns[dim];
498
499 MPI_Datatype gtype{get_mpi_type<Integer>()};
500 MPI_Gather(&starts_cpy[dim], 1, gtype, s_list[dim], 1, gtype, root, loc.topology().comm_cart);
501 MPI_Gather(&Ns[dim], 1, gtype, n_list[dim], 1, gtype, root, loc.topology().comm_cart);
502 }
503
504
505 // ------------------- Local Proc Sending to ROOT Proc ------------------- /
506 if (loc.topology().rank != root)
507 {
508 // - - - - - - begin of create sending buffer datatypes - - - - - - - - //
509 for (dim = 0; dim < dimension; ++dim)
510 {
511 array_sizes[dim] = loc.topology().__local_shape[dim];
512 array_subsizes[dim] = Ns[dim];
513 }
514
515 MPI_Type_create_subarray(dimension,
516 array_sizes.data(),
517 array_subsizes.data(),
518 array_starts.data(),
519 MPI_ORDER_C, value_type, &buf_block);
520 MPI_Type_commit(&buf_block);
521 // - - - - - - end of create sending buffer datatypes - - - - - - - - //
522
523
524 MPI_Send(loc.data() + loc.get_flat_index(indexes), 1,
525 buf_block, root, loc.topology().rank, loc.topology().comm_cart);
526
527 MPI_Type_free(&buf_block);
528 }
529
530 // // ============= ROOT Gathering Information other Processes ============= //
531 if (loc.topology().rank == root)
532 {
533 for (pid = 0; pid < num_procs; ++pid)
534 {
535 std::array<Integer, NumD> loc_starts;
536 // ------------------- At the ROOT process ------------------- //
537 if (pid == root) // Local Memory Copy
538 {
539 std::array<Integer, NumD> local_indexes;
540 for (dim = 0; dim < dimension; ++dim)
541 {
542 local_indexes[dim] = starts_cpy[dim];
543 }
544
545 std::vector<Integer> s_list_cpy[NumD];
546 std::vector<Integer> n_list_cpy[NumD];
547
548 for (size_type i = 0; i < NumD; ++i) {
549 s_list_cpy[i].resize(num_procs);
550 n_list_cpy[i].resize(num_procs);
551
552 for (size_type j = 0; j < num_procs; ++j) {
553 s_list_cpy[i][j] = s_list[i][j];
554 n_list_cpy[i][j] = n_list[i][j];
555 }
556 }
557
558 std::function<void(size_type)> copy_recursive = [&](size_type dim)
559 {
560 if (dim == NumD - 1)
561 {
562 memcpy(
563 gather.begin() + gather.get_flat_index(local_indexes),
564 loc.data() + loc.get_flat_index(local_indexes),
565 n_list_cpy[dim][pid] * sizeof(T)
566 );
567 } else {
568 for (Integer i = starts_cpy[dim]; i <= loc.topology().ends[dim] + back; ++i)
569 {
570 local_indexes[dim] = i;
571 copy_recursive(dim + 1);
572 }
573 }
574 };
575
576 copy_recursive(0);
577 }
578 // ------------------- At the Other processes ------------------- //
579 else // Recv From others
580 {
581 // - - - - - - begin of create receiving buffer datatypes - - - - - - - - //
582 for (dim = 0; dim < dimension; ++dim)
583 {
584 array_sizes[dim] = loc.topology().__global_shape[dim];
585 array_subsizes[dim] = n_list[dim][pid];
586 loc_starts[dim] = s_list[dim][pid];
587 }
588
589 MPI_Type_create_subarray(dimension,
590 array_sizes.data(),
591 array_subsizes.data(),
592 array_starts.data(),
593 MPI_ORDER_C, value_type, &buf_block);
594 MPI_Type_commit(&buf_block);
595 // - - - - - - end of create receiving buffer datatypes - - - - - - - - //
596
597 // std::cout << pid << "\t" << gather.get_flat_index(loc_starts) << std::endl;
598 MPI_Recv(gather.begin() + gather.get_flat_index(loc_starts), 1,
599 buf_block, pid, pid, loc.topology().comm_cart, MPI_STATUS_IGNORE);
600
601 MPI_Type_free(&buf_block);
602 } // end of Recv From others
603 }
604 }
