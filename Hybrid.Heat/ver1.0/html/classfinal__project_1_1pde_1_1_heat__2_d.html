Initial value:
{
[&](size_type i, size_type j) {
return out(i,j) =
this->weights[0] * (in(i-1,j) + in(i+1,j))
+ this->weights[1] * (in(i,j-1) + in(i,j+1))
+ in(i,j) * (
this->diags[0]*this->weights[0]
+ this->diags[1]*this->weights[1]
);
}
}
