179 __T&

187 __size_type index {0}, i {0};

262template <class __U, __size_type __Dims>
263std::ostream& operator<<(std::ostream& os, const __array<__U, __Dims>& in)
264{
265 // Helper Function to print multi-dimensional array
266 auto print_recursive = [&](
267 auto&& self, const __array<__U, __Dims>& arr,
268 __size_type current_dim,
269 __size_type offset) -> void
270 {
271 if (current_dim == __Dims - 1)
272 {
273os << "|";
274for (__size_type i = 0; i < arr.__shape[current_dim]; ++i)
275{
276 os << std::fixed << std::setprecision(5) << std::setw(12) << (arr.__data[offset + i]);
277}
278os << " |\n";
279 } else {
280for (__size_type i = 0; i < arr.__shape[current_dim]; ++i)
281{
282 self(self, arr, current_dim + 1, offset + i * arr.__shape.strides[current_dim]);
283}
284os << "\n";
285 }
286 }; // End of Helper Function
287
288 // Start recursive printing
289 print_recursive(print_recursive, in, 0, 0);
290
291 return os;
