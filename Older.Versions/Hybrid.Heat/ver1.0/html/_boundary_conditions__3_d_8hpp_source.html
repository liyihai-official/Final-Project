163template <typename T>
164 inline void
165 BoundaryConditions_3D<T>::SetBCinDim00(
166 Heat_3D<T> & obj,
167 const T & time)
168 {
169 auto shape_cpy { obj.in.topology().__local_shape };
170 auto starts_cpy { obj.in.topology().starts };
171
172 T x {0}, y {0}, z {0};
173
174 if (starts_cpy[0] == 1)
175 {
176 size_type i {0};
177 for (size_type j = 1; j < shape_cpy[1] - 1; ++j)
178 {
179 for (size_type k = 1; k < shape_cpy[2] - 1; ++k)
180 {
181x = ( i + starts_cpy[0] - 1) * obj.dxs[0];
182y = ( j + starts_cpy[1] - 1) * obj.dxs[1];
183z = ( k + starts_cpy[2] - 1) * obj.dxs[2];
184
185if (isDirichletBC[0])
186{
187 obj.in(i,j,k) = BCFunc[0](x,y,z,time);
188 obj.out(i,j,k) = BCFunc[0](x,y,z,time);
189
190} else if (isNeumann[0])
191{
192 obj.in(i,j,k) += BCFunc[0](x,y,z,time) * obj.dt;
193 obj.out(i,j,k) += BCFunc[0](x,y,z,time) * obj.dt;
194}
195 }
196 }
197 }
