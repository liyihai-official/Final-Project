159template <typename T>
160 inline void
161 BoundaryConditions_2D<T>::SetBCinDim00(
162 Heat_2D<T> & obj,
163 const T & time)
164 {
165 auto shape_cpy { obj.in.topology().__local_shape };
166 auto starts_cpy { obj.in.topology().starts };
167
168 T x {0}, y {0};
169
170 if (starts_cpy[0] == 1)
171 {
172 size_type i {0};
173 for (size_type j = 1; j < shape_cpy[1]-1; ++j)
174 {
175x = ( i + starts_cpy[0] - 1) * obj.dxs[0];
176y = ( j + starts_cpy[1] - 1) * obj.dxs[1];
177
178if (isDirichletBC[0])
179{
180 obj.in(i,j) = BCFunc[0](x,y,time);
181 obj.out(i,j) = BCFunc[0](x,y,time);
182}
183else if (isNeumann[0])
184{
185 obj.in(i,j) += BCFunc[0](x,y,time)*obj.dt;
186 obj.out(i,j) += BCFunc[0](x,y,time)*obj.dt;
187}
188 }
189 }
190 }
